# A "virtual" class "gDesc" underlies both "grob" and "gPath"
initGrobAutoName <- function() {
  index <- 0
  function(prefix="GRID", suffix="GROB") {
    index <<- index + 1
    paste(prefix, suffix, index, sep=".")
  }
}
grobAutoName <- initGrobAutoName()
